/** Query top tokens with where clause. */
query {
  tokens(first: 1000, orderBy: txCount, orderDirection: desc, where: { txCount_gt: 10000, volume_gte: 10000, symbol_not_in: ["USDC", "USDT", "DAI"],, name_not_contains: "Wrapped"}) {
    id
    symbol
    name
    volumeUSD
    feesUSD
    totalValueLockedUSD
    txCount
    tokenDayData(
        orderBy:date
        orderDirection:desc
        first: 1, skip: 0) {
            date
            volumeUSD
            priceUSD
            feesUSD
        }
  }
}
